97 void addTimestamp(SimulationTimestamp<SimulationTimestampPstateSet> *new_timestamp) {
98 auto hostname = new_timestamp->getContent()->getHostname();
99
100 auto hostname_search = latest_timestamps_by_host.find(hostname);
101 if (hostname_search == latest_timestamps_by_host.end()) {
102 // no pstate timestamps associated to this hostname have been added yet
103 this->trace.push_back(new_timestamp);
104 this->latest_timestamps_by_host[hostname] = this->trace.size() - 1;
105 } else {
106 // a pstate timestamp associated to this host already exists
107 SimulationTimestamp<SimulationTimestampPstateSet> *&latest_timestamp = this->trace[this->latest_timestamps_by_host[hostname]];
108
109 // if the new_timestamp has the same date as the latest_timestamp, then the new_timestamp replaces
110 // the latest_timestamp in the trace, else the new time_stamp is added to the trace and the map of latest
111 // timestamps is updated to reflect this change
112 if (std::fabs(new_timestamp->getDate() - latest_timestamp->getDate()) < DBL_EPSILON) {
113 std::swap(new_timestamp, latest_timestamp);
114
115 // now, latest_timestamp points to the contents of new_timestamp
116 // and new_timestamp points to the contents of latest_timestamp so
117 // we need to delete new_timestamp
118 delete new_timestamp;
119 } else {
120 if (new_timestamp->getDate() > latest_timestamp->getDate()) {
121 this->trace.push_back(new_timestamp);
122 this->latest_timestamps_by_host[hostname] = this->trace.size() - 1;
123 } else {
124 throw std::runtime_error(
125 "SimulationTrace<SimulationTimestampPstateSet>::addTimestamp() timestamps out of order");
126 }
127 }
128 }
129 }
